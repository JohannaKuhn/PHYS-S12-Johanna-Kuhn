<br> ## Microcontroller Code: define LED_PIN 8 define BUTTON_PIN 7 const int A1A = 3; // define pin 3 for A-1A const int A1B = 4; // define pin 4 for A-1B void setup() { pinMode(A1A, OUTPUT); // specify these pins as outputs pinMode(A1B, OUTPUT); digitalWrite(A1A, LOW); // start with the motors off digitalWrite(A1B, LOW); pinMode(LED_PIN, OUTPUT); pinMode(BUTTON_PIN, INPUT); } void loop() { if (digitalRead(BUTTON_PIN) == HIGH) { digitalWrite(LED_PIN, HIGH); // start the motor digitalWrite(A1A, HIGH); digitalWrite(A1B, LOW); delay(1); // allow the motor to run for 3 seconds } else { // stop the motor digitalWrite(A1A, LOW); // setting both pins LOW stops the motor digitalWrite(A1B, LOW); // redundant, but doesn't hurt digitalWrite(LED_PIN, LOW); delay(1); // keep the motor off for 4 seconds } }
