T = (1.0 / (A + B*log(R2/R1) )); // Calculate temperature using datasheet formula. T = T - 273.15; //Convert from Kelvin to Celcius. Serial.print("Temperature: "); Serial.print(T); Serial.println("C"); delay(500); }
